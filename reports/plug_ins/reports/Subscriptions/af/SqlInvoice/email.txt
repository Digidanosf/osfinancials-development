SEmail REGISTERS_EMAIL,

from SUBSCRIPTION_ACCOUNT a 
join debtor b on a.WAccountId = b.wAccountId
join Account c on a.WAccountId = c.WAccountId 
join dochead d on a.WAccountId = d.WAccountId 
where bEnable = 1
and d.SEXTERNALID ='%s'
