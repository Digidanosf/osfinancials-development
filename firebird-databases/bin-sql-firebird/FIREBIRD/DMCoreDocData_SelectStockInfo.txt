SELECT STOCK.WSALESACCOUNTID, 
STOCK.WCOSTACCOUNTID, 
STOCK.WSTOCKACCOUNTID, 
STOCK.WStockTypeID,
ACCOUNT2.BSUBACCOUNTS SALESSA, 
ACCOUNT1.BSUBACCOUNTS STOCKSA, 
ACCOUNT.BSUBACCOUNTS COSOFSALESA 
FROM STOCK
   LEFT OUTER JOIN  ACCOUNT ACCOUNT2 on (STOCK.WSTOCKACCOUNTID = ACCOUNT2.WACCOUNTID)
   LEFT OUTER JOIN ACCOUNT ON (STOCK.WCOSTACCOUNTID = ACCOUNT.WACCOUNTID)
   LEFT OUTER JOIN ACCOUNT ACCOUNT1 ON (STOCK.WSALESACCOUNTID = ACCOUNT1.WACCOUNTID)
where WStockid =:WStockID
