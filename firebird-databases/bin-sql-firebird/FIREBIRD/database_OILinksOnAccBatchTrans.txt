Select OILinks.*,Messages.* from OILinks  join transact on OILinks.WAmountTransactionID = 
transact.WtransactionID left join messages on (transact.WDescriptionId = Messages.WMessageId)
 where OILinks.WAccountID = :WAccountID and 
OILinks.WBatchID = :WBatchID and 
OILinks.WLinkTransactionID =:WLinkTransactionID
