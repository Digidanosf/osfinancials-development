SELECT  a.WDISCOUNTTYPEID, 
a.FVALUE, a.WSTOCKGROUPID, a.WSTOCKID, a.DSYSDATE, a.BMON, a.BTUE, a.BWED, a.BTHU, a.BFRI, a.BSAT, a.BSUN
FROM HORECAPOS_DISCOUNTS a where a.WSTOCKID = <@WSTOCKID@> 
and current_timestamp between a.DDATEFROM and a.DDATETO <@DAYSEL@>
union
SELECT  a.WDISCOUNTTYPEID, 
a.FVALUE, a.WSTOCKGROUPID, a.WSTOCKID, a.DSYSDATE, a.BMON, a.BTUE, a.BWED, a.BTHU, a.BFRI, a.BSAT, a.BSUN
FROM HORECAPOS_DISCOUNTS a where a.WSTOCKGROUPID in (0<@GROUPIDS@>0) 
and current_timestamp between a.DDATEFROM and a.DDATETO  <@DAYSEL@>
